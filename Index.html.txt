    </div>

    <script>
        // Aktualizuje zegar
        function updateClock() {
            const now = new Date();
            document.getElementById('current-time').textContent =
                `${String(now.getHours()).padStart(2, '0')}:${String(now.getMinutes()).padStart(2, '0')}`;
        }

        // Pobiera dane z poznan-mpk-api
        async function fetchDepartures(stopId) {
            const url = `https://cors-anywhere.herokuapp.com/https://poznan-mpk-api.pl/api/v1/stops/${stopId}/departures`;
            try {
                const response = await fetch(url);
                if (!response.ok) throw new Error("Błąd pobierania danych");
                return await response.json();
            } catch (error) {
                console.error("Błąd:", error);
                return null;
            }
        }

        // Przetwarza dane z API
        function processDepartures(departures) {
            if (!departures) return null;

            const now = new Date();
            return departures
                .filter(dep => dep.line === "3" || dep.line === "10") // Tylko linie 3 i 10
                .map(dep => {
                    const departureTime = new Date(dep.departureTime);
                    const diffMinutes = Math.floor((departureTime - now) / 60000);
                    return {
                        line: dep.line,
                        destination: dep.destination,
                        time: Math.max(0, diffMinutes + (dep.delay || 0)) // Uwzględnij opóźnienie
                    };
                })
                .sort((a, b) => a.time - b.time) // Sortuj po czasie
                .slice(0, 5); // Wyświetl max 5 najbliższych
        }

        // Wyświetla rozkład na stronie
        async function renderTrams() {
            const stopId = 1054; // ID przystanku "Jasna Rola"
            const scheduleElement = document.getElementById('schedule');
            scheduleElement.innerHTML = '<div class="loading">Ładowanie rozkładu...</div>';

            const departures = await fetchDepartures(stopId);
            if (!departures) {
                scheduleElement.innerHTML = '<div class="error">Nie udało się pobrać danych.</div>';
                return;
            }

            const trams = processDepartures(departures);
            if (!trams || trams.length === 0) {
                scheduleElement.innerHTML = '<div class="error">Brak odjazdów.</div>';
                return;
            }

            scheduleElement.innerHTML = '';
            trams.forEach(tram => {
                const lineClass = tram.line === "3" ? 'line-3' : 'line-10';
                const tramElement = document.createElement('div');
                tramElement.className = 'schedule-item';
                tramElement.innerHTML = `
                    <div class="line-number ${lineClass}">${tram.line}</div>
                    <div class="destination-time-container">
                        <span class="destination">${tram.destination}</span>
                        <span class="departure-time">${tram.time} min</span>
                    </div>
                `;
                scheduleElement.appendChild(tramElement);
            });
        }

        // Odświeża dane co 30 sekund
        setInterval(() => {
            updateClock();
            renderTrams();
        }, 30000);

        // Pierwsze uruchomienie
        updateClock();
        renderTrams();
    </script>
